import turtle
import random

screen = turtle.Screen()
screen.bgcolor("darkblue")
screen.title("Цветочный карнавал")
screen.setup(1000, 800)

mountain_turtle = turtle.Turtle()
mountain_turtle.speed(0)
mountain_turtle.color("gray")
mountain_turtle.penup()
mountain_turtle.goto(-500, -150)
mountain_turtle.pendown()

def draw_mountain(height, width):
    mountain_turtle.begin_fill()
    mountain_turtle.left(45)
    mountain_turtle.forward(width)
    mountain_turtle.right(90)
    mountain_turtle.forward(width)
    mountain_turtle.left(45)
    mountain_turtle.end_fill()

mountain_turtle.fillcolor("gray")
mountain_heights = [200, 250, 180, 300, 220, 270, 190, 280]
mountain_widths = [150, 120, 180, 100, 160, 140, 170, 130]

for i in range(len(mountain_heights)):
    mountain_turtle.penup()
    mountain_turtle.goto(-450 + i*110, -150)
    mountain_turtle.pendown()
    draw_mountain(mountain_heights[i], mountain_widths[i])

mountain_turtle.hideturtle()


flower_turtle = turtle.Turtle()
flower_turtle.speed(0)
flower_turtle.color("red")
flower_turtle.penup()

def draw_flower(x, y, size=15):
    flower_turtle.penup()
    flower_turtle.goto(x, y)
    flower_turtle.pendown()
    
    
    flower_turtle.color("green")
    flower_turtle.setheading(90)
    flower_turtle.forward(size * 2)
    
    
    flower_turtle.color("red")
    for _ in range(6):
        flower_turtle.forward(size)
        flower_turtle.backward(size)
        flower_turtle.right(60)
    
    
    flower_turtle.dot(size/2, "yellow")


def draw_flower2(x, y, size=15):
    flower_turtle.penup()
    flower_turtle.goto(x, y)
    flower_turtle.pendown()
    
    
    flower_turtle.color("darkgreen")
    flower_turtle.setheading(90)
    flower_turtle.forward(size * 2.5)
    
    
    flower_turtle.color("red")
    flower_turtle.begin_fill()
    flower_turtle.circle(size)
    flower_turtle.end_fill()
    
  
    flower_turtle.dot(size/1.5, "orange")

def draw_flower3(x, y, size=12):
    flower_turtle.penup()
    flower_turtle.goto(x, y)
    flower_turtle.pendown()
    
  
    flower_turtle.color("green")
    flower_turtle.setheading(90)
    flower_turtle.forward(size * 3)
    
   
    flower_turtle.color("red")
    for _ in range(8):
        flower_turtle.forward(size)
        flower_turtle.backward(size)
        flower_turtle.right(45)
    
  
    flower_turtle.dot(size/2, "yellow")

flower_types = [draw_flower, draw_flower2, draw_flower3]
flower_colors = ["red", "darkred", "crimson"]

for _ in range(50):
    x = random.randint(-450, 450)
    y = random.randint(-250, -80)
    size = random.randint(8, 20)
    flower_type = random.choice(flower_types)
    flower_turtle.color(random.choice(flower_colors))
    flower_type(x, y, size)

flower_turtle.hideturtle()
